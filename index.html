Project Overview

This project leverages the merge sort algorithm to rank urban areas based on water scarcity levels, aiming to optimize resource allocation and ensure equitable distribution.

Methodology

We implemented the merge sort algorithm to efficiently sort areas by their water scarcity index, ensuring accurate and timely rankings.

Implementation


// Sample JavaScript implementation of Merge Sort
function mergeSort(arr) {
    if (arr.length <= 1) {
        return arr;
    }
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    return merge(left, right);
}

function merge(left, right) {
    let result = [];
    let leftIndex = 0;
    let rightIndex = 0;
    while (leftIndex < left.length && rightIndex < right.length) {
        if (left[leftIndex] < right[rightIndex]) {
            result.push(left[leftIndex]);
            leftIndex++;
        } else {
            result.push(right[rightIndex]);
            rightIndex++;
        }
    }
    return result.concat(left.slice(leftIndex)).concat(right.slice(rightIndex));
}
