Project Overview

This project focuses on ranking urban areas based on water scarcity levels to optimize resource allocation and ensure equitable distribution.

Methodology

We utilize the merge sort algorithm to efficiently rank areas by their water scarcity index, ensuring accurate prioritization for resource distribution.

Implementation


// Sample JavaScript implementation of Merge Sort
function mergeSort(arr) {
    if (arr.length <= 1) {
        return arr;
    }
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    return merge(left, right);
}

function merge(left, right) {
    let result = [];
    let leftIndex = 0;
    let rightIndex = 0;
    while (leftIndex < left.length && rightIndex < right.length) {
        if (left[leftIndex].scarcityIndex < right[rightIndex].scarcityIndex) {
            result.push(left[leftIndex]);
            leftIndex++;
        } else {
            result.push(right[rightIndex]);
            rightIndex++;
        }
    }
    return result.concat(left.slice(leftIndex)).concat(right.slice(rightIndex));
}

// Example usage:
let areas = [
    { name: 'Area A', scarcityIndex: 12 },
    { name: 'Area B', scarcityIndex: 19 },
    { name: 'Area C', scarcityIndex: 3 },
    { name: 'Area D', scarcityIndex: 5 },
    { name: 'Area E', scarcityIndex: 2 }
];

let sortedAreas = mergeSort(areas);
console.log(sortedAreas);
