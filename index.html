Algorithm Selection: Merge Sort

  • Why Merge Sort:
    • Chosen for its efficiency in sorting large datasets.
    • Time complexity: O(n log n), ideal for handling extensive water scarcity data.
    • Stability: Preserves the order of equal elements, which is useful for additional sorting criteria if needed.

  • Merge Sort Algorithm (C++ Implementation):
    • Steps:
      1. Divide the dataset into smaller subsets.
      2. Sort each subset recursively.
      3. Merge the sorted subsets to produce the final sorted list.
    • C++ Code Example:
    
#include <iostream>
#include <vector>
using namespace std;

void merge(vector<int>& arr, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    vector<int> L(n1), R(n2);

    for (int i = 0; i < n1; i++)
        L[i] = arr[left + i];
    for (int j = 0; j < n2; j++)
        R[j] = arr[mid + 1 + j];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k] = L[i];
            i++;
        } else {
            arr[k] = R[j];
            j++;
        }
        k++;
    }

    while (i < n1) {
        arr[k] = L[i];
        i++;
        k++;
    }

    while (j < n2) {
        arr[k] = R[j];
        j++;
        k++;
    }
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }
}

int main() {
    vector<int> scarcity = {34, 7, 23, 32, 5, 62};
    mergeSort(scarcity, 0, scarcity.size() - 1);

    cout << "Ranked areas by water scarcity: ";
    for (int num : scarcity)
        cout << num << " ";
    return 0;
}
